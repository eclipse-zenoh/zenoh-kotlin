declarePublisher

fun declarePublisher(keyExpr: KeyExpr, qos: QoS = QoS.default(), encoding: Encoding = Encoding.default(), reliability: Reliability = Reliability.RELIABLE): <Error class: unknown class><Publisher>

Declare a Publisher on the session.

Example:

Zenoh.open(Config.default()).onSuccess {
    it.use { session ->
        "demo/kotlin/greeting".intoKeyExpr().onSuccess { keyExpr ->
            session.declarePublisher(keyExpr).onSuccess { pub ->
                pub.use {
                    println("Publisher declared on $keyExpr.")
                    var i = 0
                    while (true) {
                        val payload = "Hello for the ${i}th time!"
                        println(payload)
                        pub.put(payload)
                        Thread.sleep(1000)
                        i++
                    }
                }
            }
        }
    }
}
